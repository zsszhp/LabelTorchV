### 任务 0.4.2 — Python 训练适配器

**交付物**：`backend/labeltorch_backend/adapters/base.py` + `ultralytics_adapter.py`

```python
# base.py — 训练适配器抽象基类
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Optional

@dataclass
class TrainConfig:
    model_family: str       # "yolov8"
    model_variant: str      # "yolov8n", "yolov8s", ...
    img_size: int           # 640
    batch: int              # 16
    epochs: int             # 100
    patience: int           # 50
    device: str             # "cuda:0", "cpu"
    workers: int            # 8
    pretrained_weights: Optional[str] = None  # 路径或 "yolov8n.pt"
    amp: bool = False
    resume: bool = False

@dataclass
class TrainResult:
    success: bool
    best_weight_path: str
    last_weight_path: str
    metrics: dict           # {"mAP50": 0.95, "mAP50-95": 0.82, ...}
    log_dir: str
    epochs_completed: int
    early_stopped: bool
    error_message: Optional[str] = None

class TrainingAdapter(ABC):
    """训练适配器抽象基类"""

    @abstractmethod
    def adapter_id(self) -> str:
        """适配器唯一标识，如 'ultralytics'"""

    @abstractmethod
    def supported_families(self) -> list[str]:
        """支持的模型家族列表"""

    @abstractmethod
    def validate_config(self, config: TrainConfig) -> tuple[bool, str]:
        """校验配置，返回 (是否通过, 错误信息)"""

    @abstractmethod
    async def prepare_dataset(self, dataset_dir: str, split_ratio: float,
                               class_names: list[str], output_dir: str) -> str:
        """准备训练数据集目录结构，返回 data.yaml 路径"""

    @abstractmethod
    async def start_training(self, config: TrainConfig, data_yaml: str,
                              output_dir: str, progress_callback) -> TrainResult:
        """启动训练，通过 callback 回报进度"""

    @abstractmethod
    async def stop_training(self):
        """停止训练"""
```

```python
# ultralytics_adapter.py
class UltralyticsAdapter(TrainingAdapter):
    def adapter_id(self) -> str:
        return "ultralytics"

    def supported_families(self) -> list[str]:
        return ["yolov8"]  # V0.4 仅支持 v8，后续版本扩展

    async def start_training(self, config, data_yaml, output_dir, progress_callback):
        from ultralytics import YOLO

        model_path = config.pretrained_weights or f"{config.model_variant}.pt"
        model = YOLO(model_path)

        # 训练参数映射
        results = model.train(
            data=data_yaml,
            imgsz=config.img_size,
            batch=config.batch,
            epochs=config.epochs,
            patience=config.patience,
            device=config.device,
            workers=config.workers,
            amp=config.amp,
            project=output_dir,
            name="train",
            exist_ok=True,
            verbose=True,
        )
        # ... 解析 results 生成 TrainResult
```

### 任务 0.4.3 — Python 训练 Handler

**交付物**：`backend/labeltorch_backend/handlers/training.py`

**IPC 命令**：

#### `train.start`

请求：
```json
{
    "command": "train.start",
    "payload": {
        "run_id": "uuid",
        "dataset_dir": "/path/to/dataset",
        "split_ratio": 0.8,
        "class_names": ["defect_a", "defect_b"],
        "output_dir": "/path/to/project/models/runs/<run_id>",
        "config": {
            "model_family": "yolov8",
            "model_variant": "yolov8n",
            "img_size": 640,
            "batch": 16,
            "epochs": 100,
            "patience": 50,
            "device": "cuda:0",
            "workers": 8,
            "pretrained_weights": null,
            "amp": false
        }
    }
}
```

响应：
```json
{
    "success": true,
    "result": { "status": "started", "run_id": "uuid" }
}
```

训练期间持续发送事件：
```json
{ "event_type": "task.progress", "task_id": "uuid",
  "payload": { "epoch": 15, "total_epochs": 100,
               "metrics": { "box_loss": 0.045, "cls_loss": 0.012,
                             "mAP50": 0.82, "mAP50-95": 0.65 }}}
```

训练完成事件：
```json
{ "event_type": "task.succeeded", "task_id": "uuid",
  "payload": { "best_weight_path": "...", "last_weight_path": "...",
               "metrics": { ... }, "epochs_completed": 87, "early_stopped": true }}
```

#### `train.stop`

请求：
```json
{ "command": "train.stop", "payload": { "run_id": "uuid" } }
```

#### `train.status`

请求：
```json
{ "command": "train.status", "payload": { "run_id": "uuid" } }
```

## 4. 验收检查单

1. 选择数据集 → 选择 yolov8n → 设置 epochs=5 → 点击"开始训练"
2. 训练正常启动，日志面板实时输出
3. 进度条更新，epoch 计数正确
4. loss 曲线图实时更新
5. 训练完成（或早停），结果摘要正确显示
6. `models/runs/<run_id>/train/weights/best.pt` 文件存在
7. model_versions 表有记录，metrics 正确
8. 点击"停止"按钮，训练优雅终止
9. 训练中关闭主程序，Python 子进程不残留
10. 使用 5 张图片 + 1 epoch 冒烟测试通过
